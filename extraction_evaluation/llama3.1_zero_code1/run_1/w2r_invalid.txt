def extract_waste_transform_process_resource(text):
    w2r = {"waste": [], "transforming_process": [], "transformed_resource": []}

    for word in text.split():
        if word.lower() in ["residues", "by-products", "manure", "processing waste", "forestry residues"]:
            w2r["waste"].append(word)
        elif word.lower() in ["pre-treatment procedures", "biofuel production", "biorefineries", "valorization", "conversion methods"]:
            w2r["transforming_process"].append(word)
        elif word.lower() in ["biofuels", "biochar", "soil amendment", "microalgae", "energy crops", "biofuel"]:
            w2r["transformed_resource"].append(word)

    return w2r

w2r = extract_waste_transform_process_resource(text)
print(w2r)

The differences in active sites, electron transfer directions and bonding types during adsorption of H-kaolin and OH-kaolin were the intrinsic reasons for their distinct adsorption efficiency. Furthermore, derived chars with H-kaolin were deemed safe for application in pollutant adsorption in industrial field and soil improvement in gardens or parks.
