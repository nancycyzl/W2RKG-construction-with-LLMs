def extract_resources(text: str) -> dict:
    w2r = {"waste": [], "transforming_process": [], "transformed_resource": []}

    pattern = r"(biomass and )?(waste|material)"
    matches = re.findall(pattern, text)
    for match in matches:
        if len(match) == 2:  # To account for 'biomass and' prefix
            w2r["waste"].append(match[1])

    pattern = r"MAP (pyrolysis|conversion)"
    matches = re.findall(pattern, text)

    pattern = r"(products produced )?(biomass and )?waste"
    matches = re.findall(pattern, text)
    for match in matches:
        resource = ' '.join(match).replace("waste", "product") if "waste" in match else ' '.join(match).replace("biomass and ", "")
        w2r["transformed_resource"].append(resource)
    
    return w2r
